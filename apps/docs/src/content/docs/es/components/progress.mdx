import { Tabs, TabItem } from '@astrojs/starlight/components'

import { Progress as VProgress } from '@moderno/vue'
import { Progress as SProgress } from '@moderno/svelte'
import { Progress as OProgress } from '@moderno/solid'

**Progress** muestra el estado de avance de una tarea — barra lineal o anillo circular,
determinado (un porcentaje conocido) o indeterminado (duración desconocida). Primitive
**closed-prop** (ADR-0003) sobre la máquina `progress` de Zag.js.

## Uso

<Tabs syncKey="framework">
<TabItem label="React">
```tsx
import { Progress } from '@moderno/react'

<Progress label="Subiendo" value={40} />
<Progress label="Conectando" value={null} variant="circular" />
```
</TabItem>
<TabItem label="Vue">
```vue
<script setup lang="ts">
import { Progress } from '@moderno/vue'

<template>
  <Progress label="Subiendo" v-model="subido" />
</template>
```
</TabItem>

## Props

| Prop | Tipo | Default | Descripción |
|------|------|---------|-------------|
| `label` | `string` | — | Etiqueta visible. |
| `defaultValue` | `number \| null` | `50` | Valor inicial no controlado. `null` inicia indeterminado. |
| `value` / `v-model` / `bind:value` | `number \| null` | — | Valor controlado. `null` renderiza una barra/anillo indeterminado. |
| `onValueChange` | `(value: number \| null) => void` | — | Callback al cambiar. |
| `min` | `number` | `0` | Valor mínimo. |
| `max` | `number` | `100` | Valor máximo. |
| `variant` | `'linear' \| 'circular'` | `'linear'` | Renderiza como barra o anillo. |
| `showValue` | `boolean` | `true` | Muestra el texto de valor formateado junto a la etiqueta. |
