import { Menu as OMenu } from '@moderno/solid'

export const actions = [
  { label: 'Edit', value: 'edit' },
  { label: 'Duplicate', value: 'duplicate' },
  { label: 'Archive', value: 'archive' },
  { label: 'Delete', value: 'delete' },
]

**Menu** is a pull-down of actions: a trigger button that opens a list of items, each
firing `onSelect` with its `value`. A **closed-prop** primitive (ADR-0003) built on
Zag.js's `menu` machine — open/close, keyboard navigation, typeahead, focus return, and
ARIA `menu`/`menuitem` come from Zag; Moderno owns the markup and the `md-menu-*` class
contract. For picking a form value, use `Select` or `Combobox` instead.

## Usage

<Tabs syncKey="framework">
<TabItem label="React">
```tsx
import { Menu } from '@moderno/react'

const actions = [
  { label: 'Edit', value: 'edit' },
  { label: 'Delete', value: 'delete' },
]

<Menu triggerLabel="Actions" items={actions} onSelect={(value) => console.log(value)} />
```
</TabItem>

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `triggerLabel` | `string` | — | Label of the trigger button. |
| `items` | `{ label: string; value: string; disabled?: boolean }[]` | — | The actions offered by the menu. |
| `onSelect` / `@select` | `(value: string) => void` | — | Called with the item's `value` when selected. |
