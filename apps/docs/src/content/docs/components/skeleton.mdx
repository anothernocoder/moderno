import { Tabs, TabItem } from '@astrojs/starlight/components'
import { Skeleton as RSkeleton } from '@moderno/react'
import { Skeleton as VSkeleton } from '@moderno/vue'
import { Skeleton as SSkeleton } from '@moderno/svelte'
import { Skeleton as OSkeleton } from '@moderno/solid'

**Skeleton** is a loading placeholder with a shimmer animation. It is decorative (`aria-hidden`), pure CSS. The animations respect `prefers-reduced-motion`.

## Usage

<Tabs syncKey="framework">
<TabItem label="React">
```tsx
import { Skeleton } from '@moderno/react'

<Skeleton variant="circle" width={48} height={48} />
<Skeleton variant="text" width="80%" />
```
</TabItem>
<TabItem label="Vue">
```vue
<script setup lang="ts">
import { Skeleton } from '@moderno/vue'
</script>

<template>
  <Skeleton variant="circle" :width="48" :height="48" />
  <Skeleton variant="text" width="80%" />
</template>
```
</TabItem>
<TabItem label="Svelte">
```svelte
<script lang="ts">
  import { Skeleton } from '@moderno/svelte'
</script>

<Skeleton variant="circle" width={48} height={48} />
<Skeleton variant="text" width="80%" />
```
</TabItem>
<TabItem label="Solid">
```tsx
import { Skeleton } from '@moderno/solid'

<Skeleton variant="circle" width={48} height={48} />
<Skeleton variant="text" width="80%" />
```
</TabItem>
</Tabs>

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `'text' \| 'rect' \| 'circle'` | `'text'` | Shape of the placeholder. |
| `width` | `string \| number` | — | CSS width; a number is interpreted as px. |
| `height` | `string \| number` | — | CSS height; a number is interpreted as px. |
